104  if (option_reuse_address())

106 #if (defined(unix) || defined(__unix) || defined(__unix__) || defined(__APPLE__)) && !defined(__CYGWIN__)

111  }
112 #endif
113  if (option_multicast())
114  _socket.bind(_endpoint);
115  else
116  _socket.bind(asio::ip::udp::endpoint(_endpoint.protocol(), 0));
117 
118  // Prepare receive buffer

121  // Reset statistic
